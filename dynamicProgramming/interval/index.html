{

        return;
    if (dp[L][R] != 0)
        return;

    return;
}

    int c[N], dp[N][N];

int f(int L, int R)
{
    if (dp[L][R] != -1)
        return dp[L][R];
    if (L == R - 1)
        return dp[L][R] = 0;
    dp[L][R] = INT_MAX;
    for (int i = L + 1; i < R; ++i)
        dp[L][R] = min(dp[L][R], f(L, i) + f(i, R) + (c[R] - c[L]));
    return dp[L][R];
}

    int f(int L, int R)
{
    if (L == R)
        return dp[L][R] = 0;
    if (dp[L][R] != -1)
        return dp[L][R];
    dp[L][R] = INT_MAX;
    for (int i = L; i < R; ++i)
    {
        int res = f(L, i) + f(i + 1, R) + x[L] * x[i + 1] * x[R + 1];
        if(res < dp[L][R])
        {
            dp[L][R] = res;
            to[L][R] = i;
        }
    }
    return dp[L][R];
}
