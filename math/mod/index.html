    T pow(int a, int b, int c)
{ // calculate a^b%c
    T res = 1, tmp = a;
    for (; b; b >>= 1)
    {
        if (b & 1)
            res = res * tmp % c;
        tmp = tmp * tmp % c;
    }
    return res;
}

int extgcd(int a, int b, int c, int &x, int &y)
{
    int d = a;
    if (b)
    {
        d = extgcd(b, a % b, c, y, x);
        y -= (a / b) * x;
    }
    else
    {
        x = c;
        y = 0;
    }
    return d;
}
